28 /* Priority of the rule added by the fail-open subsystem when a switch enters
29  * fail-open mode. This priority value uniquely identifies a fail-open flow
30  * (OpenFlow priorities max out at 65535 and nothing else in Open vSwitch
31  * creates flows with this priority). And "f0" is mnemonic for "fail open"! */
32 #define FAIL_OPEN_PRIORITY 0xf0f0f0

34 /* Returns true if 'rule' is one created by the "fail open" logic, false
35  * otherwise. */
36 static inline bool
37 is_fail_open_rule(const struct rule *rule)
38 {
39  return rule->cr.priority == FAIL_OPEN_PRIORITY;
40 }
