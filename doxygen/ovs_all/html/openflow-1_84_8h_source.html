392 /* Body for ofp14_multipart_request of type OFPMP_FLOW_MONITOR.
393  *
394  * The OFPMP_FLOW_MONITOR request's body consists of an array of zero or more
395  * instances of this structure. The request arranges to monitor the flows
396  * that match the specified criteria, which are interpreted in the same way as
397  * for OFPMP_FLOW.
398  *
399  * 'id' identifies a particular monitor for the purpose of allowing it to be
400  * canceled later with OFPFMC_DELETE. 'id' must be unique among
401  * existing monitors that have not already been canceled.
402  */
403 struct ofp14_flow_monitor_request {
404  ovs_be32 monitor_id; /* Controller-assigned ID for this monitor. */
405  ovs_be32 out_port; /* Required output port, if not OFPP_ANY. */
406  ovs_be32 out_group; /* Required output port, if not OFPG_ANY. */
407  ovs_be16 flags; /* OFPMF14_*. */
408  uint8_t table_id; /* One table's ID or OFPTT_ALL (all tables). */
409  uint8_t command; /* One of OFPFMC14_*. */
410  /* Followed by an ofp11_match structure. */
411 };
412 OFP_ASSERT(sizeof(struct ofp14_flow_monitor_request) == 16);

426  OFPFMF14_ADD = 1 << 1, /* New matching flows as they are added. */
427  OFPFMF14_REMOVED = 1 << 2, /* Old matching flows as they are removed. */
428  OFPFMF14_MODIFY = 1 << 3, /* Matching flows as they are changed. */
429 
430  /* What to include in updates. */
431  /* Common to NX and OpenFlow 1.4 */
432  OFPFMF14_INSTRUCTIONS = 1 << 4, /* If set, instructions are included. */
433  OFPFMF14_NO_ABBREV = 1 << 5, /* If set, include own changes in full. */
434  /* OpenFlow 1.4 */
435  OFPFMF14_ONLY_OWN = 1 << 6, /* If set, don't include other controllers.
436  */
