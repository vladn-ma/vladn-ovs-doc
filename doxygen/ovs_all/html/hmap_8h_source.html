107 /* Search.
108  *
109  * HMAP_FOR_EACH_WITH_HASH iterates NODE over all of the nodes in HMAP that
110  * have hash value equal to HASH. HMAP_FOR_EACH_IN_BUCKET iterates NODE over
111  * all of the nodes in HMAP that would fall in the same bucket as HASH. MEMBER
112  * must be the name of the 'struct hmap_node' member within NODE.
113  *
114  * These macros may be used interchangeably to search for a particular value in
115  * an hmap, see, e.g. shash_find() for an example. Usually, using
116  * HMAP_FOR_EACH_WITH_HASH provides an optimization, because comparing a hash
117  * value is usually cheaper than comparing an entire hash map key. But for
118  * simple hash map keys, it makes sense to use HMAP_FOR_EACH_IN_BUCKET because
119  * it avoids doing two comparisons when a single simple comparison suffices.
120  *
121  * The loop should not change NODE to point to a different node or insert or
122  * delete nodes in HMAP (unless it "break"s out of the loop to terminate
123  * iteration).
124  *
125  * HASH is only evaluated once.
126  *
127  * When the loop terminates normally, meaning the iteration has completed
128  * without using 'break', NODE will be NULL. This is true for all of the
129  * HMAP_FOR_EACH_*() macros.
130  */
131 #define HMAP_FOR_EACH_WITH_HASH(NODE, MEMBER, HASH, HMAP) \
132  for (INIT_CONTAINER(NODE, hmap_first_with_hash(HMAP, HASH), MEMBER); \
133  (NODE != OBJECT_CONTAINING(NULL, NODE, MEMBER)) || (NODE = NULL); \
134  ASSIGN_CONTAINER(NODE, hmap_next_with_hash(&(NODE)->MEMBER), \
135  MEMBER))
136 #define HMAP_FOR_EACH_IN_BUCKET(NODE, MEMBER, HASH, HMAP) \
137  for (INIT_CONTAINER(NODE, hmap_first_in_bucket(HMAP, HASH), MEMBER); \
138  (NODE != OBJECT_CONTAINING(NULL, NODE, MEMBER)) || (NODE = NULL); \
139  ASSIGN_CONTAINER(NODE, hmap_next_in_bucket(&(NODE)->MEMBER), MEMBER))

205 /* Inserts 'node', with the given 'hash', into 'hmap'. 'hmap' is never
206  * expanded automatically. */
207 static inline void
208 hmap_insert_fast(struct hmap *hmap, struct hmap_node *node, size_t hash)
209 {
210  struct hmap_node **bucket = &hmap->buckets[hash & hmap->mask];
211  node->hash = hash;
212  node->next = *bucket;
213  *bucket = node;
214  hmap->n++;
215 }

217 /* Inserts 'node', with the given 'hash', into 'hmap', and expands 'hmap' if
218  * necessary to optimize search performance.
219  *
220  * ('where' is used in debug logging. Commonly one would use hmap_insert() to
221  * automatically provide the caller's source file and line number for
222  * 'where'.) */
223 static inline void
224 hmap_insert_at(struct hmap *hmap, struct hmap_node *node, size_t hash,
225  const char *where)

227  hmap_insert_fast(hmap, node, hash);

233 /* Removes 'node' from 'hmap'. Does not shrink the hash table; call
234  * hmap_shrink() directly if desired. */
235 static inline void
236 hmap_remove(struct hmap *hmap, struct hmap_node *node)
237 {
238  struct hmap_node **bucket = &hmap->buckets[node->hash & hmap->mask];
239  while (*bucket != node) {
240  bucket = &(*bucket)->next;
241  }
242  *bucket = node->next;
243  hmap->n--;
244 }

246 /* Puts 'new_node' in the position in 'hmap' currently occupied by 'old_node'.
247  * The 'new_node' must hash to the same value as 'old_node'. The client is
248  * responsible for ensuring that the replacement does not violate any
249  * client-imposed invariants (e.g. uniqueness of keys within a map).
250  *
251  * Afterward, 'old_node' is not part of 'hmap', and the client is responsible
252  * for freeing it (if this is desirable). */
253 static inline void
254 hmap_replace(struct hmap *hmap,
255  const struct hmap_node *old_node, struct hmap_node *new_node)
256 {
257  struct hmap_node **bucket = &hmap->buckets[old_node->hash & hmap->mask];
258  while (*bucket != old_node) {
259  bucket = &(*bucket)->next;
260  }
261  *bucket = new_node;
262  new_node->hash = old_node->hash;
263  new_node->next = old_node->next;
264 }
