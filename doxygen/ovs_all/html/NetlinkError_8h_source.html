204 static __inline
205 NlMapStatusToNlErr(NTSTATUS status)
206 {
207  NL_ERROR ret;
208 
209  switch (status)
210  {
211  case STATUS_NOT_SUPPORTED:
212  ret = NL_ERROR_NOTSUPP;
213  break;
214  case STATUS_INSUFFICIENT_RESOURCES:
215  ret = NL_ERROR_NOMEM;
216  break;
217  case STATUS_SUCCESS:
218  ret = NL_ERROR_SUCCESS;
219  break;
220  case STATUS_PENDING:
221  ret = NL_ERROR_PENDING;
222  break;
223  case STATUS_CANCELLED:
224  ret = NL_ERROR_CANCELED;
225  break;
226  case STATUS_INVALID_PARAMETER:
227  ret = NL_ERROR_INVAL;
228  break;
229  case STATUS_OBJECT_NAME_EXISTS:
230  ret = NL_ERROR_EXIST;
231  break;
232  default:
233  ret = NL_ERROR_OTHER;
234  break;
235  }
236 
237  return ret;
238 }
