26 /* A flow classification match.
27  *
28  * Use one of the match_*() functions to initialize a "struct match".
29  *
30  * The match_*() functions below maintain the following important invariant.
31  * If a bit or a field is wildcarded in 'wc', then the corresponding bit or
32  * field in 'flow' is set to all-0-bits. (The match_zero_wildcarded_fields()
33  * function can be used to restore this invariant after adding wildcards.) */
34 struct match {
35  struct flow flow;
36  struct flow_wildcards wc;
37  struct tun_metadata_allocation tun_md;
38 };

161 /* A sparse representation of a "struct match".
162  *
163  * 'flows' is used for allocating both 'flow' and 'mask' with one
164  * miniflow_alloc() call.
165  *
166  * There are two invariants:
167  *
168  * - The same invariant as "struct match", that is, a 1-bit in the 'flow'
169  * must correspond to a 1-bit in 'mask'.
170  *
171  * - 'flow' and 'mask' have the same 'map'. This implies that 'flow' and
172  * 'mask' have the same part of "struct flow" at the same offset into
173  * 'values', which makes minimatch_matches_flow() faster.
174  */
175 struct minimatch {
176  union {
177  struct {
178  struct miniflow *flow;
179  struct minimask *mask;
180  };
181  struct miniflow *flows[2];
182  };
183 };
