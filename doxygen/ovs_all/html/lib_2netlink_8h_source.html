20 /* Netlink message helpers.
21  *
22  * Netlink is a datagram-based network protocol primarily for communication
23  * between user processes and the kernel, and mainly on Linux. Netlink is
24  * specified in RFC 3549, "Linux Netlink as an IP Services Protocol".
25  *
26  * Netlink is not suitable for use in physical networks of heterogeneous
27  * machines because host byte order is used throughout.
28  *
29  * This header file defines helper functions for working with Netlink messages.
30  * For Netlink protocol definitions, see netlink-protocol.h. For
31  * Linux-specific definitions for Netlink sockets, see netlink-socket.h.
32  */

96 /* Sizes of various attribute types, in bytes, including the attribute header
97  * and padding.
98  *
99  * A minimum-size attribute is 4 bytes long: 4 bytes of header, no bytes of
100  * payload, no padding.
101  *
102  * A maximum-size attribute is 65536 bytes long: 4 bytes of header, 65531 bytes
103  * of payload, 1 byte of padding. (Thus, NL_ATTR_SIZE() of a maximum length
104  * attribute payload does not fit in 16 bits.) */
105 #define NL_ATTR_SIZE(PAYLOAD_SIZE) (NLA_HDRLEN + NLA_ALIGN(PAYLOAD_SIZE))
106 #define NL_A_U8_SIZE NL_ATTR_SIZE(sizeof(uint8_t))
107 #define NL_A_U16_SIZE NL_ATTR_SIZE(sizeof(uint16_t))
108 #define NL_A_U32_SIZE NL_ATTR_SIZE(sizeof(uint32_t))
109 #define NL_A_U64_SIZE NL_ATTR_SIZE(sizeof(uint64_t))
110 #define NL_A_BE16_SIZE NL_ATTR_SIZE(sizeof(ovs_be16))

158 /* This macro is careful to check for attributes with bad lengths. */
159 #define NL_ATTR_FOR_EACH(ITER, LEFT, ATTRS, ATTRS_LEN) \
160  for ((ITER) = (ATTRS), (LEFT) = (ATTRS_LEN); \
161  nl_attr_is_valid(ITER, LEFT); \
162  (LEFT) -= nl_attr_len_pad(ITER, LEFT), (ITER) = nl_attr_next(ITER))
163 
164 
165 /* This macro does not check for attributes with bad lengths. It should only
166  * be used with messages from trusted sources or with messages that have
167  * already been validated (e.g. with NL_ATTR_FOR_EACH). */
168 #define NL_ATTR_FOR_EACH_UNSAFE(ITER, LEFT, ATTRS, ATTRS_LEN) \
169  for ((ITER) = (ATTRS), (LEFT) = (ATTRS_LEN); \
170  (LEFT) > 0; \
171  (LEFT) -= nl_attr_len_pad(ITER, LEFT), (ITER) = nl_attr_next(ITER))
