36 /* Buffer for holding arbitrary data. An ofpbuf is automatically reallocated
37  * as necessary if it grows too large for the available memory.
38  *
39  * 'header' and 'msg' conventions:
40  *
41  * OpenFlow messages: 'header' points to the start of the OpenFlow
42  * header, while 'msg' is the OpenFlow msg bofy.
43  * When parsing, the 'data' will move past these, as data is being
44  * pulled from the OpenFlow message.
45  *
46  * Caution: buffer manipulation of 'struct ofpbuf' must always update
47  * the 'header' and 'msg' pointers.
48  *
49  *
50  * Actions: When encoding OVS action lists, the 'header' is used
51  * as a pointer to the beginning of the current action (see ofpact_put()).
52  *
53  * rconn: Reuses 'header' as a private pointer while queuing.
54  */
55 struct ofpbuf {
56  void *base; /* First byte of allocated space. */
57  void *data; /* First byte actually in use. */
58  uint32_t size; /* Number of bytes in use. */
59  uint32_t allocated; /* Number of bytes allocated. */
60 
61  void *header; /* OpenFlow header. */
62  void *msg; /* message's body */
63  struct ovs_list list_node; /* Private list element for use by owner. */
64  enum ofpbuf_source source; /* Source of memory allocated as 'base'. */
65 };

67 /* An initializer for a struct ofpbuf that will be initially empty and
68  * uses the space in STUB (which should be an array) as a stub.
69  *
70  * Usage example:
71  *
72  * uint64_t stub[1024 / 8]; // 1 kB stub properly aligned for 64-bit data.
73  * struct ofpbuf ofpbuf = OFPBUF_STUB_INITIALIZER(stub);
74  */
75 #define OFPBUF_STUB_INITIALIZER(STUB) { \
76  .base = (STUB), \
77  .data = (STUB), \
78  .size = 0, \
79  .allocated = sizeof (STUB), \
80  .header = NULL, \
81  .msg = NULL, \
82  .list_node = OVS_LIST_POISON, \
83  .source = OFPBUF_STUB, \
84  }

150 /* If 'b' contains at least 'offset + size' bytes of data, returns a pointer to
151  * byte 'offset'. Otherwise, returns a null pointer. */
152 static inline void *ofpbuf_at(const struct ofpbuf *b, size_t offset,
153  size_t size)
154 {
155  return offset + size <= b->size ? (char *) b->data + offset : NULL;
156 }

158 /* Returns a pointer to byte 'offset' in 'b', which must contain at least

173 /* Returns a pointer to byte following the last byte allocated for use (but

180 /* Returns the number of bytes of headroom in 'b', that is, the number of bytes
181  * of unused space in ofpbuf 'b' before the data that is in use. (Most
182  * commonly, the data in a ofpbuf is at its beginning, and thus the ofpbuf's
183  * headroom is 0.) */
184 static inline size_t ofpbuf_headroom(const struct ofpbuf *b)
185 {
186  return (char*)b->data - (char*)b->base;
187 }
188 
189 /* Returns the number of bytes that may be appended to the tail end of ofpbuf
190  * 'b' before the ofpbuf must be reallocated. */
191 static inline size_t ofpbuf_tailroom(const struct ofpbuf *b)
192 {
193  return (char*)ofpbuf_end(b) - (char*)ofpbuf_tail(b);
194 }

203 /* Removes 'size' bytes from the head end of 'b', which must contain at least
204  * 'size' bytes of data. Returns the first byte of data removed. */
205 static inline void *ofpbuf_pull(struct ofpbuf *b, size_t size)
206 {
207  void *data = b->data;
208  b->data = (char*)b->data + size;
209  b->size = b->size - size;
210  return data;
211 }
212 
213 /* If 'b' has at least 'size' bytes of data, removes that many bytes from the
214  * head end of 'b' and returns the first byte removed. Otherwise, returns a
215  * null pointer without modifying 'b'. */
216 static inline void *ofpbuf_try_pull(struct ofpbuf *b, size_t size)
217 {
218  return b->size >= size ? ofpbuf_pull(b, size) : NULL;
219 }
