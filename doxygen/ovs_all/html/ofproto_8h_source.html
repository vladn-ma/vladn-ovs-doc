366 /* The behaviour of the port regarding VLAN handling */
367 enum port_vlan_mode {
368  /* This port is an access port. 'vlan' is the VLAN ID. 'trunks' is
369  * ignored. */
370  PORT_VLAN_ACCESS,
371 
372  /* This port is a trunk. 'trunks' is the set of trunks. 'vlan' is
373  * ignored. */
374  PORT_VLAN_TRUNK,
375 
376  /* Untagged incoming packets are part of 'vlan', as are incoming packets
377  * tagged with 'vlan'. Outgoing packets tagged with 'vlan' stay tagged.
378  * Other VLANs in 'trunks' are trunked. */
379  PORT_VLAN_NATIVE_TAGGED,
380 
381  /* Untagged incoming packets are part of 'vlan', as are incoming packets
382  * tagged with 'vlan'. Outgoing packets tagged with 'vlan' are untagged.
383  * Other VLANs in 'trunks' are trunked. */
384  PORT_VLAN_NATIVE_UNTAGGED
385 };
